	    'date' 	=> 'date_format:Y-m-d',
	    'password' 	=> 'confirmed|min:6|max:8|regex:/^(?=.*?[A-Z])(?=.*?[a-z])(?=.*?[0-9])(?=.*?[#?!@$%^&*-]).{6,}$/',

	]);
